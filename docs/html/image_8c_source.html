58 static int image_read(struct blkdev *dev, int offset, int len, void *buf)
59 {
60  struct image_dev *im = dev->private;
61 
62  /* to fail a disk we close its file descriptor and set it to -1 */
63  if (im->fd == -1) {
64  return E_UNAVAIL;
65  }
66  assert(offset >= 0 && offset+len <= im->nblks);
67 
68  ssize_t result = pread(im->fd, buf, len*BLOCK_SIZE, offset*BLOCK_SIZE);
69 
70  /* Since I'm not asking for the code that calls this to handle
71  * errors other than E_BADADDR and E_UNAVAIL, we report errors and
72  * then exit. Since we already checked the address, this shouldn't
73  * happen very often.
74  */
75  if (result < 0) {
76  fprintf(stderr, "read error on %s: %s\n", im->path, strerror(errno));
77  assert(0);
78  }
79 
80  if (result != len*BLOCK_SIZE) {
81  fprintf(stderr, "short read on %s: %s\n", im->path, strerror(errno));
82  assert(0);
83  }
84 
85  return SUCCESS;
86 }

168 struct blkdev *image_create(char *path)
169 {

171  struct image_dev *im = malloc(sizeof(*im));
172 
173  if (dev == NULL || im == NULL)
174  return NULL;
175 
176  im->path = strdup(path); /* save a copy for error reporting */
177 
178  /* open image device */
179  im->fd = open(path, O_RDWR);
180  if (im->fd < 0) {
181  fprintf(stderr, "can't open image %s: %s\n", path, strerror(errno));
182  return NULL;
183  }
184 
185  /* access image device */
186  struct stat sb;
187  if (fstat(im->fd, &sb) < 0) {
188  fprintf(stderr, "can't access image %s: %s\n", path, strerror(errno));
189  return NULL;
190  }
191 
192  /* print a warning if file is not a multiple of the block size -
193  * this isn't a fatal error, as extra bytes beyond the last full
194  * block will be ignored by read and write.
195  */
196  if (sb.st_size % BLOCK_SIZE != 0) {
197  fprintf(stderr, "warning: file %s not a multiple of %d bytes\n",
198  path, BLOCK_SIZE);
199  }
200  im->nblks = sb.st_size / BLOCK_SIZE;
201  dev->private = im;
202  dev->ops = &image_ops;
203 
204  return dev;
205 }
