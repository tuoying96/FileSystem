85 static void _strmode(int mode, char *buf)
86 {
87  char *str = "rwxrwxrwx";
88  if (S_ISDIR(mode)) *buf++ = 'd';
89  else if (S_ISLNK(mode)) *buf++ = 'l';
90  else *buf++ = '-';
91  for (int mask = 0400; mask != 0; str++, mask = mask >> 1) {
92  *buf++ = (mask & mode) ? *str : '-';
93  }
94  *buf++ = ' '; // last character will always be space
95  *buf++ = 0;
96 }

168 static int normalize_dot_paths(char *names[], int nnames) {
169  int dest = 0;
170  for (int src = 0; src < nnames; src++) {
171  if (strcmp(names[src],".") == 0) {
172  // ignore current directory path element
173  free(names[src]); // "." no longer used
174  names[src] = NULL;
175  } else if (strcmp(names[src], "..") == 0) {
176  if (dest > 0) {
177  dest--; // back up one directory element
178  free(names[dest]); // names[dest] no longer used
179  names[dest] = NULL;
180  }
181  free(names[src]); // ".." no longer used
182  names[src] = NULL;
183  } else { // add directory element
184  if (dest < src) {
185  names[dest] = names[src];
186  names[src] = NULL;
187  }
188  dest++;
189  }
190  }
191  return dest;
192 }

204  // make full path for path

209  if (pathlen == 0) {

364 int do_ls1(char *argv[])
365 {
366  struct stat sb;
367  char pathbuf[PATH_MAX];
368  full_path(argv[0], pathbuf);
369 
370  // Appending "/." to path causes getattr to resolve
371  // symlinks in the path except the leaf, and return
372  // the stat buffer for the real directory.
373  char pathdotbuf[PATH_MAX];
374  strcpy(pathdotbuf, pathbuf);
375  strcat(pathdotbuf, (pathbuf[strlen(pathbuf)-1] == '/') ? "." : "/.");
376  int retval = fs_ops.getattr(pathdotbuf, &sb);
377  if (retval == 0) { // directory
378  struct fuse_file_info info;
379  memset(&info, 0, sizeof(struct fuse_file_info));
380  if ((retval = fs_ops.opendir(pathbuf, &info)) == 0) {
381  init_ls(pathbuf);
382  /* read directory information */
383  retval = fs_ops.readdir(pathbuf, NULL, filler, 0, &info);
384  }
385  fs_ops.releasedir(pathbuf, &info);
386  } else {
387  // use the actual path to see if it is a file or symlink
388  retval = fs_ops.getattr(pathbuf, &sb);
389  if (retval == 0) {
390  init_ls("/"); // pathbuf is full path so root relative
391  /* list file information */
392  retval = filler(NULL, pathbuf, &sb, 0);
393  }
394  }
395  if (retval == 0) {
396  /* print file or directory information */
397  print_ls();
398  }
399

663 static int do_get(char *argv[])
664 {
665  char *outside = argv[0], *inside = argv[1];
666  char path[PATH_MAX];
667  int len, fd, offset = 0, val;
668 
669  if ((fd = open(outside, O_RDONLY, 0)) < 0) {
670  return fd;
671  }
672  full_path(inside, path);
673  if ((val = fs_ops.mknod(path, 0777, 0)) != 0) {
674  close(fd);
675  return val;
676  }
677 
678  struct fuse_file_info info;
679  memset(&info, 0, sizeof(struct fuse_file_info));
680  if ((val = fs_ops.open(path, &info)) != 0) {
681  return val;
682  }
683  while ((len = read(fd, blkbuf, blksiz)) > 0) {
684  val = fs_ops.write(path, blkbuf, len, offset, &info);
685  if (val != len) {
686  break;
687  }
688  offset += len;
689  }
690  close(fd);
691  fs_ops.release(path, &info);
692  return (val >= 0) ? 0 : val;
693 }
