43 {
44  uint32_t buf[PTRS_PER_BLK];
45 
46  // get entry from direct blocks
47  struct fs_inode *in = &fs.inodes[inum];
48  if (n < N_DIRECT) {
49  if (in->direct[n] == 0) {
50  if (alloc == 0) {
51  return 0;
52  }
53  int blkno = get_free_blk();
54  if (blkno == 0) { // no space
55  return 0;
56  }
57  in->direct[n] = blkno;
58  mark_inode(inum);
59  disk->ops->write(disk, in->direct[n], 1, zeros);
60  }
61  return in->direct[n];
62  }
63 
64  // get entry from single-indirect block
65  n -= N_DIRECT;
66  if (n < PTRS_PER_BLK) {
67  if (in->indir_1 == 0) {
68  if (alloc == 0) {
69  return 0;
70  }

72  int blkno = get_free_blk();

74  return 0;
75  }

78  disk->ops->write(disk, in->indir_1, 1, zeros);
79  }

83  return 0;
84  }

87  if (blkno == 0) { // no space

200 int do_read(int inum, char* buf, size_t len, off_t offset) {
201  // get pointer to inode for inum
202  struct fs_inode *in = &fs.inodes[inum];
203 
204  // done if offset greater than file size
205  if (offset >= in->size) {
206  return 0;
207  }
208 
209  // adjust length to length of file from offset
210  if (in->size < offset + len) {
211  len = in->size - offset;
212  }
213 
214  // index of first block
215  int blkidx1 = offset / FS_BLOCK_SIZE;
216  int blkidx2 = (offset + len) / FS_BLOCK_SIZE;
217 
218  // read blocks into buf
219  offset -= blkidx1 * FS_BLOCK_SIZE;
220  int _len = len;
221  for (int blkindex = blkidx1; blkindex <= blkidx2; blkindex++) {
222  // get block for block index
223  char blk[FS_BLOCK_SIZE];
224  int blkno = get_file_blk(inum, blkindex, blk, 0);
225 
226  // report error if not found
227  if (blkno <= 0) {
228  return -EIO;
229  }
230 
231  // copy block content to buf
232  int l = min(FS_BLOCK_SIZE - offset, len);
233  memcpy(buf, &blk[offset], l);
234 
235  buf += l;
236  len -= l;
237  offset = 0;
238  }
239 
240  return _len;
241 
242 }

262 int do_write(int inum, const char* buf, size_t len, off_t offset) {
263  // get pointer to inode for inum
264  struct fs_inode *in = &fs.inodes[inum];
265 
266  // return error code of offset out of range
267  if (offset > in->size) {
268  return -EINVAL;
269  }
270 
271  int blkidx1 = offset / FS_BLOCK_SIZE;
272  int blkidx2 = (offset + len) / FS_BLOCK_SIZE;
273 
274  // write buffer to file blocks
275  offset -= blkidx1 * FS_BLOCK_SIZE;
276  int _len = len;
277  for (int blkindex = blkidx1; blkindex <= blkidx2; blkindex++) {
278  // get block
279  char blk[FS_BLOCK_SIZE];
280  int blkno = get_file_blk(inum, blkindex, blk, 1);
281 
282  // return error code if error or out of space
283  if (blkno <= 0) {
284  mark_inode(inum);
285  flush_metadata();
286  return (blkno == 0) ? -ENOSPC : blkno;
287  }
288 
289  // copy more buffer fuse_parser_data to block
290  int l = min(FS_BLOCK_SIZE - offset, len);
291  memcpy(&blk[offset], buf, l);
292 
293  // write block back to disk
294  disk->ops->write(disk, blkno, 1, blk);
295  buf += l;
296  len -= l;
297  offset = 0;
298  in->size += l;
299  in->mtime = time(NULL); // OK thorough 2100
300  }
301 
302  mark_inode(inum);
303  flush_metadata();
304 
305  return _len - len;

319 int do_truncate(int inum, int len)
320 {
321  if (len != 0) {
322  return -EINVAL; /* invalid argument */
323  }
324 
326  struct fs_inode *in = &fs.inodes[inum];
327 
328  uint32_t buf[PTRS_PER_BLK], buf1[PTRS_PER_BLK];
329  int i, j;
330 
331  /* unlink double indirect nodes */
332  if (in->indir_2) {
333  disk->ops->read(disk, in->indir_2, 1, buf);
334  for (i = 0; i < PTRS_PER_BLK; i++) {
335  if (buf[i] != 0) { // block allocated
336  disk->ops->read(disk, buf[i], 1, buf1);
337  for (j = 0; j < PTRS_PER_BLK; j++) {
338  if (buf1[j] != 0) { // block allocated
339  return_blk(buf1[j]); // free block
340  }
341  }
342  return_blk(buf[i]); // return head block
343  }
344  }
345  return_blk(in->indir_2); // free head block
346  in->indir_2 = 0;
347  }
348 
349  /* unlink single indirect nodes */
350  if (in->indir_1) {
351  disk->ops->read(disk, in->indir_1, 1, buf);
352  for (i = 0; i < PTRS_PER_BLK; i++) {
353  if (buf[i] != 0) {
354  return_blk(buf[i]); // free head block
355  }
356  }
357  return_blk(in->indir_1);
358  in->indir_1 = 0;
359  }
360 
361  /* unlink direct nodes */
362  for (i = 0; i < N_DIRECT; i++) {
363  if (in->direct[i] != 0) {
364  return_blk(in->direct[i]); // free direct blocks
365  in->direct[i] = 0;
366  }
367  }
368 
369  // reset inode size and modification time
370  in->size = 0;
371  in->mtime = time(NULL); // OK thorough 2100
372 
373  return 0;
