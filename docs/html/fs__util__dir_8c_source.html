127 int get_dir_entry_block(
128  int inum, void *block, int* blkno, const char* name)
129 {
130  // ensure that inode for inum is a directory
131  if (!S_ISDIR(fs.inodes[inum].mode)) {
132  *blkno = 0; // no block
133  memset(block, 0, FS_BLOCK_SIZE);
134  return -ENOTDIR;
135  }
136 
137  // get first block of directory
138  int dir_blkno = get_file_blk(inum, 0, block, 0); // no extend
139  if (dir_blkno <= 0) {
140  *blkno = 0; // no block
141  memset(block, 0, FS_BLOCK_SIZE);
142  return (dir_blkno == 0) ? -ENOENT : -EIO;
143  }
144 
145  // find entry in block
146  int entry_no = get_dir_entry_in_block(block, name);
147  if (entry_no >= 0) {
148  *blkno = dir_blkno;
149  } else {
150  *blkno = 0; // no block
151  memset(block, 0, FS_BLOCK_SIZE);
152  }
153  return entry_no;
154 }

253 int do_rmdir(int dir_inum, const char* leaf) {
254  // report error if inode not a directory
255  struct fs_inode *din = &fs.inodes[dir_inum];
256  if (!S_ISDIR(din->mode)) {
257  return -ENOTDIR;
258  }
259 
261  int blkno;
262  char buf[FS_BLOCK_SIZE];
263  int entno = get_dir_entry_block(dir_inum, buf, &blkno, leaf);
264  if (entno < 0) {
265  return -ENOENT; // entry not found
266  }
267 
268  // get inode of directory entry
269  struct fs_dirent *de = (void*)buf;
270  int entry_inum = de[entno].inode;
271 
272  // ensure that entry being removed is a directory
273  if (!S_ISDIR(fs.inodes[entry_inum].mode)) {
274  return -ENOTDIR; // entry must be directory
275  }
276 
277  // ensure directory being removed is empty
278  // 0 indicates not empty, < 0 indicates error
279  if (is_dir_empty(entry_inum) != 1) {
280  return -ENOTEMPTY;
281  }
282 
283  // mark directory inode free and flush its block
284  de[entno].valid = 0;
285  disk->ops->write(disk, blkno, 1, buf);
286 
287  // truncate all blocks of unlinked directory inode
288  do_truncate(entry_inum, 0);
