51 static void init_path_state(struct path_state* ps, const char* path, int noleaf) {
52  ps->inum = fs.root_inode;
53  ps->symcount = 0;
54  ps->idx = 0;

56  // add '.' to end of path if ends with PATH_DELIM
57  char lpath[strlen(path)+2];
58  strcpy(lpath, path);
59  int idx = strlen(path) - strlen(PATH_DELIM);
60  if (idx >= 0) {

68  ps->pathtoks = malloc(ps->pathlen*sizeof(char*));

109 static int get_inode_of_path_state(struct path_state* ps)
110 {
111  int inum = ps->inum; // initial inum
112  int status;
113  // status == 0 for normal termination, < 0 for error
114  while ((status = next_path_state(ps, inum)) > 0) {
115  // resolve next symlink path element
116  status = 0;
117  inum = ps->inum; // accept inode of current token
118  ps->idx++; // advance to next path token
119  }
120 
121  // return inum or error status
122  return (status < 0) ? status : inum;
123 }
